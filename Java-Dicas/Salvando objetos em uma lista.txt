public void readJTable(){

    DefaultTableModel modelo = (DefaultTableModel) jTProdutos.getModel();

    ProdutoDAO pdao = new ProdutoDAO();
    
    for(Produto p: pdao.read()){

        modelo.addRow(new Object[]{
            p.getId(),
            p.getDescricao(),
            p.getQtd(),
            p.getPreco()
        });

    }

}
